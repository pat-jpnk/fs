# fs
A command line utility program

fs stands for file search

build: 

  - "cmake --build ." in /build


help:
- "./fs" in /build <br/>

examples:

> search -s [search term] <br/>
> ./fs -s "superman"

> info -i <br/>
> ./fs -i

> replace -r [target term] [replacement term] <br/>
> -/fs -r "superman" "man"
